#include <bits/stdc++.h>

using namespace std;

int solution(vector<int> order) {
    int answer = 0;
    int box[order.size()];
    
    for(int i=0; i<order.size(); i++) {
        box[order[i]-1] = i;
    }
    
    int index = 0;
    stack<int> s;
    
    for(int b : box) {
        if(b == answer) {
            answer++;
        } else {
            s.push(b);
        }
        while(!s.empty() && s.top() == answer) {
            answer++;
            s.pop();
        }
    }
    
    return answer;
}
